using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;

public class ButtonListener: MonoBehaviour {
    public Button m_YourButton;
    public Text m_MyText;
    public double counter;

  // Use this for initialization
	void Start () {
        Button btn = m_YourButton.GetComponent<Button>();
        btn.onClick.AddListener(TaskOnClick);
    }
	
    void TaskOnClick()
    {
        Debug.Log("Button clicked!");

        var androidJC = new AndroidJavaClass("com.unity3d.player.UnityPlayer");
        var javaObject = androidJC.GetStatic<AndroidJavaObject>("currentActivity");
        
        //Will pause Unity from Android activity
        javaObject.CallStatic("callMe", "Call to Android");
    }
    
    // Update is called once per frame
    void Update () {
        //Counter to show that Unity in the back runs, is paused and runs again
        counter++;
        m_MyText.text = counter.ToString();
	}
}
